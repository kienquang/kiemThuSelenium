"""
Automated Test: Product Sorting by Price
Website: https://icondenim.com/collections/tat-ca-san-pham
Framework: Python 3 + Selenium WebDriver (Chrome)

Cấu trúc HTML đã xác minh qua DevTools:
    <div class="box-pro-prices">
        <p class="pro-price">
            <span> 199,000đ ~ 398,000đ </span>   <!-- 1 span duy nhất, có thể là dải giá -->
        </p>
        <div class="product-sale">
            <span>-40%</span>                     <!-- badge % nằm NGOÀI p.pro-price -->
        </div>
    </div>

Quy tắc lấy giá:
    - Lấy text của: div.box-pro-prices p.pro-price span  (1 span duy nhất)
    - Nếu text chứa "~" → là dải giá → tách lấy giá nhỏ (sort tăng) hoặc giá lớn (sort giảm)
    - Badge % nằm trong div.product-sale, KHÔNG ảnh hưởng vì selector đã đúng scope
"""

import re
import sys
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait, Select
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import (
    TimeoutException,
    NoSuchElementException,
    StaleElementReferenceException,
    WebDriverException,
)

# ---------------------------------------------------------------------------
# Constants
# ---------------------------------------------------------------------------

URL = "https://icondenim.com/collections/tat-ca-san-pham"
PAGE_LOAD_TIMEOUT = 30

SORT_DROPDOWN_CSS = "select.sort-by.custom-dropdown__select"
PRODUCT_LIST_CSS  = "div.content-product-list.product-list.filter.d-flex.row-left-list"
PRODUCT_BLOCK_CSS = "div.product-block.item"

# Selector chính xác theo DevTools: span duy nhất bên trong p.pro-price
PRICE_SPAN_CSS    = "div.box-pro-prices p.pro-price span"

PRICE_ASC_KEYWORDS  = ["price-ascending",  "price asc",  "low to high",  "tăng dần"]
PRICE_DESC_KEYWORDS = ["price-descending", "price desc", "high to low",  "giảm dần"]


# ---------------------------------------------------------------------------
# 1. Mở trình duyệt
# ---------------------------------------------------------------------------

def open_website() -> webdriver.Chrome:
    """Khởi động Chrome, điều hướng đến URL và chờ danh sách sản phẩm tải xong."""
    options = Options()
    options.add_argument("--headless=new")
    options.add_argument("--no-sandbox")
    options.add_argument("--disable-dev-shm-usage")
    options.add_argument("--window-size=1920,1080")

    driver = webdriver.Chrome(options=options)
    driver.maximize_window()

    print(f"[INFO] Đang mở: {URL}")
    driver.get(URL)

    WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(
        EC.presence_of_element_located((By.CSS_SELECTOR, PRODUCT_LIST_CSS))
    )
    print("[INFO] Trang đã tải thành công.\n")
    return driver


# ---------------------------------------------------------------------------
# 2. Lấy dropdown sắp xếp
# ---------------------------------------------------------------------------

def get_sort_dropdown(driver: webdriver.Chrome) -> Select:
    """Tìm thẻ <select> sắp xếp và trả về đối tượng Select của Selenium."""
    el = WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(
        EC.presence_of_element_located((By.CSS_SELECTOR, SORT_DROPDOWN_CSS))
    )
    return Select(el)


# ---------------------------------------------------------------------------
# 3. Chọn một tuỳ chọn sắp xếp
# ---------------------------------------------------------------------------

def select_sort_option(driver: webdriver.Chrome, select_obj: Select, index: int) -> str:
    """Chọn tuỳ chọn theo index, trả về tên hiển thị của tuỳ chọn."""
    option       = select_obj.options[index]
    option_text  = option.text.strip()
    option_value = option.get_attribute("value") or ""
    print(f"[ACTION] Chọn [{index}]: '{option_text}' (value='{option_value}')")
    select_obj.select_by_index(index)
    return option_text


# ---------------------------------------------------------------------------
# 4. Chờ sắp xếp hoàn tất
# ---------------------------------------------------------------------------

def wait_for_sort_completed(driver: webdriver.Chrome, prev_first_price: str) -> None:
    """
    Poll cho đến khi text của span giá đầu tiên thay đổi so với trước khi sort.
    Fallback: nếu không phát hiện thay đổi (timeout), chỉ cần đảm bảo
    container sản phẩm vẫn hiển thị (sort giống nhau / trang không đổi).
    """
    wait = WebDriverWait(driver, PAGE_LOAD_TIMEOUT, poll_frequency=0.4,
                         ignored_exceptions=[StaleElementReferenceException])
    try:
        wait.until(lambda d: _get_first_price_text(d) != prev_first_price)
    except TimeoutException:
        WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(
            EC.presence_of_element_located((By.CSS_SELECTOR, PRODUCT_LIST_CSS))
        )


def _get_first_price_text(driver: webdriver.Chrome) -> str:
    """Lấy text thô của span giá đầu tiên, dùng để phát hiện thay đổi sau sort."""
    els = driver.find_elements(By.CSS_SELECTOR, PRICE_SPAN_CSS)
    return els[0].text.strip() if els else ""


# ---------------------------------------------------------------------------
# 5. Lấy danh sách giá — ĐÃ SỬA ĐÚNG
# ---------------------------------------------------------------------------

def _parse_price_from_text(raw: str, category: str) -> int | None:
    """
    Chuyển đổi text giá thành số nguyên.

    Xử lý 2 trường hợp:
      1. Giá đơn:  "499,000đ"         → 499000
      2. Dải giá:  "199,000đ ~ 398,000đ"
            - sort tăng dần (price_asc)  → lấy giá NHỎ: 199000
            - sort giảm dần (price_desc) → lấy giá LỚN: 398000
            - khác                       → lấy giá NHỎ (mặc định)

    Trả về None nếu không parse được.
    """
    raw = raw.strip()
    if not raw:
        return None

    if "~" in raw:
        # Tách thành [giá_nhỏ, giá_lớn]
        parts = raw.split("~")
        if len(parts) != 2:
            return None
        # Giảm dần → đại diện bằng giá lớn nhất trong dải
        target = parts[1].strip() if category == "price_desc" else parts[0].strip()
    else:
        target = raw

    # Xoá ký tự không phải số: đ, ₫, dấu chấm, dấu phẩy, khoảng trắng
    cleaned = re.sub(r"[^\d]", "", target)
    return int(cleaned) if cleaned.isdigit() else None


def get_prices(driver: webdriver.Chrome, category: str) -> list[int]:
    prices = []
    
    # Sử dụng hằng số đã định nghĩa thay vì khai báo lại cục bộ
    WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(
        EC.presence_of_element_located((By.CSS_SELECTOR, PRODUCT_BLOCK_CSS))
    )
    
    product_blocks = driver.find_elements(By.CSS_SELECTOR, PRODUCT_BLOCK_CSS)
    
    for block in product_blocks:
        if block.is_displayed():
            try:
                price_el = block.find_element(By.CSS_SELECTOR, PRICE_SPAN_CSS)
                text = price_el.text.strip()
                
                # Bỏ qua nếu không chứa số
                if not any(char.isdigit() for char in text):
                    continue
                    
                # Gọi hàm parse đã định nghĩa ở trên thay vì tự xử lý lại
                numeric_price = _parse_price_from_text(text, category)
                
                if numeric_price is not None:
                    prices.append(numeric_price)
                    
            except NoSuchElementException:
                # Bỏ qua các block lỗi DOM hoặc không có thẻ giá
                continue
            except Exception as e:
                print(f"[DEBUG] Lỗi khi lấy giá: {e}")
                continue
                
    return prices

# ---------------------------------------------------------------------------
# 6 & 7. Hàm kiểm tra thứ tự
# ---------------------------------------------------------------------------

def is_ascending(prices: list[int]) -> bool:
    """Kiểm tra danh sách tăng dần (low → high)."""
    return prices == sorted(prices)


def is_descending(prices: list[int]) -> bool:
    """Kiểm tra danh sách giảm dần (high → low)."""
    return prices == sorted(prices, reverse=True)


def _classify_option(option_text: str, option_value: str) -> str:
    """
    Phân loại tuỳ chọn sắp xếp.
    Trả về: 'price_asc' | 'price_desc' | 'other' | 'skip'
    """
    if not option_value:
        return "skip"
    combined = (option_text + " " + option_value).lower()
    if any(kw in combined for kw in PRICE_ASC_KEYWORDS):
        return "price_asc"
    if any(kw in combined for kw in PRICE_DESC_KEYWORDS):
        return "price_desc"
    return "skip"


# ---------------------------------------------------------------------------
# 8. Đánh giá một test case
# ---------------------------------------------------------------------------

def run_test(category: str, prices: list[int]) -> str:
    """Đánh giá danh sách giá theo loại sort và trả về kết quả test."""
    if category == "skip":
        return "SKIP"
    if not prices:
        return "FAIL (không tìm thấy giá)"
    if category == "price_asc":
        return "PASS" if is_ascending(prices) else "FAIL"
    if category == "price_desc":
        return "PASS" if is_descending(prices) else "FAIL"
    return "SKIP"


# ---------------------------------------------------------------------------
# 9. Hàm điều phối chính
# ---------------------------------------------------------------------------

def main() -> None:
    """Khởi động browser, lặp qua tất cả tuỳ chọn sort, in kết quả từng case."""
    driver = None
    summary: list[tuple[str, str]] = []

    try:
        driver = open_website()

        sort_select = get_sort_dropdown(driver)
        total       = len(sort_select.options)
        print(f"[INFO] Tìm thấy {total} tuỳ chọn sắp xếp.\n")

        for idx in range(total):
            print("=" * 50)
            try:
                # Lấy lại Select mỗi vòng để tránh StaleElementReferenceException
                sort_select  = get_sort_dropdown(driver)
                option       = sort_select.options[idx]
                option_text  = option.text.strip()
                option_value = option.get_attribute("value") or ""
                category     = _classify_option(option_text, option_value)

                if category == "skip":
                    print(f"[SKIP] [{idx}] '{option_text}' — không có value, bỏ qua.\n")
                    summary.append((option_text or f"Option[{idx}]", "SKIP"))
                    continue

                # Ghi nhận giá đầu tiên trước khi sort (để detect thay đổi)
                prev_price = _get_first_price_text(driver)

                # Chọn tuỳ chọn sắp xếp
                option_text = select_sort_option(driver, sort_select, idx)

                # Chờ danh sách cập nhật
                wait_for_sort_completed(driver, prev_price)

                # Lấy danh sách giá (truyền category để xử lý dải giá đúng chiều)
                prices = get_prices(driver)

                # Đánh giá
                result = run_test(category, prices)

                # In kết quả
                print(f"Tuỳ chọn sắp xếp : {option_text}")
                print(f"Danh sách giá    : {prices}")
                print(f"Kết quả          : {result}\n")

                summary.append((option_text, result))

            except (TimeoutException, NoSuchElementException) as exc:
                msg = f"FAIL ({type(exc).__name__})"
                print(f"[ERROR] [{idx}] — {exc}\n")
                summary.append((f"Option[{idx}]", msg))

            except Exception as exc:
                msg = f"FAIL (lỗi không mong muốn: {type(exc).__name__})"
                print(f"[ERROR] [{idx}] — {exc}\n")
                summary.append((f"Option[{idx}]", msg))

    except WebDriverException as exc:
        print(f"[FATAL] Không thể khởi động Chrome: {exc}")
        sys.exit(1)

    finally:
        if driver:
            driver.quit()
            print("[INFO] Đã đóng trình duyệt.")

    # Tổng kết
    print("\n" + "=" * 50)
    print("TỔNG KẾT KẾT QUẢ KIỂM THỬ")
    print("=" * 50)
    passed = failed = skipped = ni = 0
    for label, res in summary:
        if res == "PASS":
            icon = "✅"; passed += 1
        elif res.startswith("FAIL"):
            icon = "❌"; failed += 1
        elif res == "SKIP":
            icon = "⏭️"; skipped += 1
        else:
            icon = "⚠️"; ni += 1
        print(f"  {icon}  {label:<40} → {res}")

    print("-" * 50)
    print(f"  PASS: {passed}  |  FAIL: {failed}  |  SKIP: {skipped}  |  CHƯA CÀI ĐẶT: {ni}")
    print("=" * 50)


if __name__ == "__main__":
    main()
